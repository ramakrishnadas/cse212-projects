public static string findIndexDuplicate(string myString) 
{
    var mySet = new HashSet<char>();
    for (int i = 0; i < myString.Length; i++)
    {
        if (!mySet.Add(myString[i])) 
        {
            return $"The first letter that is duplicated is {myString[i]} at index {i}";
        }
    }
    return "";
}
